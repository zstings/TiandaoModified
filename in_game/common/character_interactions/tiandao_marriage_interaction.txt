# 脚本参考来源: https://eu5.paradoxwikis.com/Character_modding
tiandao_marriage_interaction = {
    sound = UI_action_character_marry_noble       # 沿用游戏结婚音效
    on_own_nation = yes                           # 只能对自己国家的角色使用
    context_menu_click_mode = click               # 直接生效，无确认弹窗
    message = yes                                 # 弹消息提示
    
    # 决定按钮在谁的右键菜单显示
    potential = {
        # 只有玩家看得见这个按钮
        scope:actor = { is_player = yes }
        # 这里不需要在这里判断年龄，因为我们在选择器里过滤
    }

    # 步骤 1：选择第一个人（被赐婚者）
    select_trigger = {
        looking_for_a = character
        source = actor # 列出玩家国家的所有角色
        target_flag = recipient # 将第一个人存为 recipient
        name = "tiandao_marriage_interaction_choose_first_character"
        column = { data = name }
        visible = {
            is_alive = yes
            is_married = no
            age_in_years >= 17
            age_in_years <= 30
        }
        enabled = {
            is_alive = yes
        }
    }

    # 步骤 2：选择第二个人（结婚对象）
    select_trigger = {
        # top_widget = marry_noble_character_top # 如果你想用官方的UI样式可以开启
        looking_for_a = character
        source = actor # 列出玩家国家的所有角色
        target_flag = target # 将第二个人存为 target
        name = "tiandao_marriage_interaction_choose_first_character"
        column = { data = character_info } # 列出角色完整信息
        
        visible = {
            is_alive = yes
            is_married = no
            age_in_years >= 17
            age_in_years <= 30
            # 逻辑过滤
            NOT = { is_same_gender = scope:recipient } # 性别相反
            NOT = { this = scope:recipient }           # 不能是同一个人
            NOT = { is_spouse_of = scope:recipient }   # 还没结婚
            NOT = { is_close_relative = scope:recipient } # 排除近亲（官方指令）
            NOT = { dynasty = scope:recipient.dynasty } #同姓排除
            NOT = { has_estate = estate_type:burghers_estate }#排除市民
            # 官方安全判定
            # 官方提供的综合判定，包含了角色是否在监狱、是否因某些 modifier 无法结婚等隐藏条件
            character_can_marry_trigger = yes
            modifier:blocked_from_character_interactions = no
        }
    }

    # 步骤 3：执行效果
    effect = {
        if = {
            limit = {
                exists = scope:recipient
                exists = scope:target
            }
            # 使用官方结婚效果指令[cite: 4]
            scope:recipient = {
                marry_character = scope:target
            }
        }
    }

    ai_will_do = { value = 0 }
}
