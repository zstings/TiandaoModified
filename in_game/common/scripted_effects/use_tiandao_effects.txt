# 定义一个名为 input_number_digit 的“函数”
# 使用 $DIGIT$ 作为参数占位符
input_number_digit = {
    change_global_variable = {
        name = pinyin_code
        multiply = 10
    }
    change_global_variable = {
        name = pinyin_code
        add = $DIGIT$
    }
}

# 定义一个生成角色的 effect，方便 GUI 调用
generate_character_create_effect = {
    create_named_dynasty = $dynasty$
    create_character = {
        adm = $adm$
        dip = $dip$
        mil = $mil$
        dynasty = dynasty:$dynasty$
        first_name = $first_name$
        age = $age$
        female = $sex$
        # female = yes         # 性别
        culture = root.culture  # 继承国家的文化
        religion = root.religion # 继承国家的宗教
        estate = estate_type:nobles_estate # 贵族
    }
}

# 定义常遇春家族 (修正了年龄，并添加了长子常茂)
generate_character_create_effect_changyucun = {
    create_named_dynasty = 常
    create_named_dynasty = 蓝
    create_character = {
        adm = { 50 80 }
        dip = { 40 70 }
        mil = { 75 99 } # 常遇春是猛将，军事极高
        dynasty = dynasty:常
        first_name = 遇春
        age = 25
        female = no
        culture = root.culture
        religion = root.religion
        estate = estate_type:nobles_estate
        save_scope_as = chang_yuchun_scope
    }
    # 妻子蓝氏（玉娘）
    create_character = {
        adm = { 40 70 }
        dip = { 40 70 }
        mil = { 30 60 } # 将门虎女
        dynasty = dynasty:蓝 # 设定为蓝氏家族，暗示与蓝玉的关系
        first_name = 玉娘 # 给她一个代号，或者用 "氏"
        age = 23 # 比常遇春小两岁
        female = yes
        culture = root.culture
        religion = root.religion
        estate = estate_type:nobles_estate
        # 保存为变量，以便生成儿子时指定母亲
        save_scope_as = chang_wife_scope
    }
    scope:chang_wife_scope = {
        set_spouse = scope:chang_yuchun_scope
    }
    # 长子常茂，生于1355年，此时为婴儿
    create_character = {
        adm = { 30 60 }
        dip = { 30 60 }
        mil = { 30 60 }
        dynasty = dynasty:常
        first_name = 茂
        age = 0
        female = no
        culture = root.culture
        religion = root.religion
        estate = estate_type:nobles_estate
        father = scope:chang_yuchun_scope # 指定父亲
        mother = scope:chang_wife_scope # 指定母亲
    }
    #蓝玉
    create_character = {
        adm = { 50 80 }
        dip = { 40 70 }
        mil = { 70 95 } # 蓝玉也是顶级猛将
        dynasty = dynasty:蓝
        first_name = 玉
        age = 18 # 蓝玉比常遇春小很多，此时是少年将领
        female = no
        culture = root.culture
        religion = root.religion
        estate = estate_type:nobles_estate
    }
    # ✅ 用 marry_character 设置夫妻关系（双向）
    scope:chang_yuchun_scope = {
        marry_character = scope:chang_wife_scope
    }
    # 兄弟姐妹不能直接指定，需要指定两个角色的共同父母来实现
}
