# 定义一个名为 input_number_digit 的“函数”
# 使用 $DIGIT$ 作为参数占位符
input_number_digit = {
    change_global_variable = {
        name = pinyin_code
        multiply = 10
    }
    change_global_variable = {
        name = pinyin_code
        add = $DIGIT$
    }
}

# 定义一个生成角色的 effect，方便 GUI 调用
generate_character_create_effect = {
    create_named_dynasty = $dynasty$
    create_character = {
        adm = $adm$
        dip = $dip$
        mil = $mil$
        dynasty = dynasty:$dynasty$
        first_name = $first_name$
        age = $age$
        female = $sex$
        # female = yes         # 性别
        culture = root.culture  # 继承国家的文化
        religion = root.religion # 继承国家的宗教
        estate = estate_type:nobles_estate # 贵族
    }
}

# 内置家族间相互通婚
tiandao_auto_marry = {
    random_character = {
        limit = {
            is_alive = yes
            is_female = no
            age_in_years >= 17
            age_in_years <= 19
            is_married = no
            OR = {
                dynasty = dynasty:徐
                dynasty = dynasty:常
                dynasty = dynasty:汤
                dynasty = dynasty:刘
                dynasty = dynasty:李
                dynasty = dynasty:沐
                dynasty = dynasty:傅
            }
        }
        save_scope_as = groom # 锁定男方
        #debug_log = xx33xx[GetPlayer.GetName]
        if = {
            limit = {
                scope:groom = { exists = yes }
            }
            root = {
                random_character = {
                    limit = {
                        is_alive = yes
                        is_female = yes
                        age_in_years >= 17
                        age_in_years <= 19
                        is_married = no
                        # 排除近亲
                        NOT = { is_close_relative = scope:groom }
                        NOT = { dynasty = scope:groom.dynasty }
                        OR = {
                            dynasty = dynasty:徐
                            dynasty = dynasty:常
                            dynasty = dynasty:汤
                            dynasty = dynasty:刘
                            dynasty = dynasty:李
                            dynasty = dynasty:沐
                            dynasty = dynasty:傅
                        }
                    }
                    marry_character = scope:groom
                    #debug_log = xx44xx[GetPlayer.GetName]
                }
            }
        }
    }
}
